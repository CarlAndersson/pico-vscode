136#define CYW43_REASON_PRUNE_CIPHER_NA (12) // BSS's cipher not supported
137#define CYW43_REASON_PRUNE_KNOWN_STA (13) // AP is already known to us as a STA
138#define CYW43_REASON_PRUNE_CCXFAST_DROAM (14) // CCX FAST ROAM: prune unqualified AP
